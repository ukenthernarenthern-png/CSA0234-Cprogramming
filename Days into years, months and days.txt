#include <stdio.h>

int main() {
    int total_days, years, months, days;

    printf("Enter total number of days: ");
    scanf("%d", &total_days);

    years = total_days / 365;
    total_days = total_days % 365;

    months = total_days / 30;
    days = total_days % 30;

    printf("Years = %d\n", years);
    printf("Months = %d\n", months);
    printf("Days = %d\n", days);

    return 0;
}
